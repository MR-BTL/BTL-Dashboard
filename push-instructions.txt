# After creating your Personal Access Token, run these commands:

# 1. Remove the old remote
git remote remove btl

# 2. Add the remote with your GitHub username and token
# Replace YOUR_USERNAME with your GitHub username (e.g., MR-BTL)
# Replace YOUR_TOKEN with the token you just created
git remote add btl https://YOUR_USERNAME:YOUR_TOKEN@github.com/MR-BTL/BTL-Dashboard.git

# 3. Push to GitHub
git push btl main

# OR if using your own repository:
git remote set-url origin https://YOUR_USERNAME:YOUR_TOKEN@github.com/EyadOraby/dashboard.git
git push origin main
